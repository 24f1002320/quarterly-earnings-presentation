Example: Data extraction snippet

// fetch earnings JSON and compute revenue growth
async function fetchEarnings(ticker) {
  const res = await fetch(`/api/earnings/${ticker}.json`);
  const data = await res.json();
  return data;
}

const { current, prior } = await fetchEarnings('ACME');
const growth = ((current.revenue - prior.revenue) / prior.revenue) * 100;
console.log(`Revenue growth: ${growth.toFixed(2)}%`);
This snippet demonstrates how backend data can feed the slides or dashboard.
